A free, functional, MVVM-oriented GUI framework for rapid prototyping of desktop applications written in Rust, supported by the Mozilla WebRender rendering engine.

Features

  • Model-View-View-Model (MVVM) programming model
  • CSS 2.1-based CSS engine, support for many common CSS properties
  • SVG / 2D drawing helpers
  • Flexbox-based layout description
  • DOM-based stateless components
  • Stateless dynamic CSS properties
  • Async I/O helpers
  • OpenGL integration
  • Huge library of more than 50 common-use GUI widgets
  • Enables custom widgets via function composition
  • Single deployment binary, minimal binary size, CPU and RAM usage
  • Minimal redraw time (0.5 - 4ms), efficient state caching

Hello world

extern crate azul;
use azul::prelude::*;
use azul::widgets::*;
struct MyDataModel {
/* your data model here */
counter: usize,
}
impl Layout for MyDataModel {
// note that the data model is immutable here
fn layout(&self, _window_id: WindowId) -> Dom<Self> {
let button = Dom::new(NodeType::Widget(
button(ButtonContent::Text("Update counter"))
.with_id("my_awesome_button")
.with_callback(On::MouseUp, update_counter)
));
let label = Dom::new(NodeType::Label(format!("{}", self.counter)))
.with_id("the_counter")
));
Dom::new(NodeType::Div)
.with_child(button)
.with_child(label)
}
}
// Callback that is called when the button was clicked
fn update_counter(app_state: &mut AppState<MyDataModel>, _event: WindowEvent) -> UpdateScreen {
app_state.data.modify(|data| data.counter += 1;);
UpdateScreen::Redraw
}
fn main() {
let css = Css::new_from_string("body { flex-direction: column; }").unwrap();
let mut app = App::new(MyDataModel { counter: 0 });
app.create_window(WindowCreateOptions::default(), css).unwrap();
app.run();
}
