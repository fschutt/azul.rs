Cross-platform MIT-licensed desktop GUI framework for C and Rust, backed by the Mozilla WebRender rendering engine.

#include "azul.h" struct DataModel { counter: uint32_t } StyledDom renderMyDataModel(RefAny* data, LayoutInfo info) { AzStyledDom view = AzStyledDom_default(); DataModel* data = AzRefAny_downcastRef(data); if !(data) { return view; } AzLabel label = AzString_format("{}", &[AzInt_new(data.counter)]); AzButton button = AzString_static("Update counter"); AzRefAny applicationDataRefCopy = AzRefAny_shallowCopy(data); AzButton_setOnMouseUp(customOnButtonClick, applicationDataRefCopy); AzStyledDom_append(AzLabel_new(AzLabel_dom(label))); AzStyledDom_append(AzButton_dom(AzButton_withLabel(button))); return view; } UpdateScreen customOnButtonClick(RefAny* data, CallbackInfo event) { DataModel data = AzRefAny_downcastMut(data); if !(data) { return DoNothing; } data.counter += 1; return RegenerateDomForCurrentWindow; } int main() { AzRefAny applicationData = AzRefAny_new(DataModel { .counter = 0 }); AzApp app = AzApp_new(application_data, AzAppConfig_default()); AzApp_run(app, AzWindowCreateOptions_new(renderMyDataModel)) }

Azul can render infinitely large datasets (such as a table, shown here) while using a comparably small amount of memory (Qt: 123Mb, Electron: 400Mb). Users can specify special IFrameCallbacks that - given the size and scroll position of the DOM node - only return the currently visible cells. All cells can be styled via CSS, but share their default style via a *const CssPropertyVec - so that the CSS properties don't get duplicated in memory.
